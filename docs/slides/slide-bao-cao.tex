% ---------- Heuristics ----------
\begin{frame}{Hàm heuristic đề xuất}
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{block}{$h_1(s)$: Số đối tượng chưa đến đích}
\[
  h_1(s) = (1{-}W) + (1{-}G) + (1{-}C)
\]
\begin{itemize}
  \item Đếm số đối tượng $\{W,G,C\}$ còn ở bờ trái.
  \item Phạm vi: $0 \leq h_1 \leq 3$
  \item \textbf{Admissible}: mỗi đối tượng cần $\geq 1$ chuyến.
  \item Tính toán $O(1)$, đơn giản.
\end{itemize}
\textcolor{Primary}{\textit{Ví dụ:}} $(F{=}0,\,W{=}0,\,G{=}1,\,C{=}0)$\\
$\Rightarrow h_1 = (1{-}0)+(1{-}1)+(1{-}0) = 2$
\end{block}

\column{0.50\textwidth}
\begin{block}{$h_2(s)$: Ước lượng số chuyến tối thiểu}
\[
  h_2(s) = \begin{cases}
    2k - 1 & F=0,\ k > 0 \\
    1       & F=0,\ k = 0 \\
    2k + 1  & F=1,\ k > 0 \\
    0       & F=1,\ k = 0
  \end{cases}
\]
{\scriptsize $k = h_1(s)$ = số đối tượng còn ở bờ trái}
\begin{itemize}
  \item Tính đến vị trí thuyền (bờ nào F đang đứng).
  \item \textbf{Admissible + Consistent}: không bao giờ ước lượng thừa.
  \item \textbf{Informed hơn} $h_1$: bao giờ cũng $h_2 \geq h_1$.
\end{itemize}
\textcolor{Primary}{\textit{Ví dụ:}} $(F{=}1,\,W{=}0,\,G{=}0,\,C{=}0)$\\
$k=3,\ F=1 \Rightarrow h_2 = 2{\times}3+1 = 7$
\end{block}
\end{columns}

% ---------- Algorithms ----------
\begin{frame}{Giới thiệu giải thuật sử dụng}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{block}{Nhóm giải thuật}
\begin{itemize}
  \item \textbf{Uninformed Search:} BFS, DFS -- không dùng heuristic
  \item \textbf{Informed Search:} A*, IDA* -- dùng $f(s)=g(s)+h(s)$
\end{itemize}
\end{block}
\vspace{0.2cm}
\begin{block}{So sánh đặc điểm}
\begin{tabular}{lcccc}
\toprule
 & \textbf{BFS} & \textbf{DFS} & \textbf{A*} & \textbf{IDA*} \\
\midrule
Tối ưu & \textcolor{Good}{\checkmark} & \textcolor{red!70!black}{$\times$} & \textcolor{Good}{\checkmark} & \textcolor{Good}{\checkmark} \\
Bộ nhớ & Cao & Thấp & Cao & Thấp \\
Heuristic & Không & Không & $h_2$ & $h_2$ \\
\bottomrule
\end{tabular}
\end{block}
